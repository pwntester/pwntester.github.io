def is_printable(s):  
    for ch in s:

            return False
    return True

def flip_byte(msg, pos, byte=0x10):  
    msg = msg[:pos] + chr(ord(msg[pos]) ^ byte) + msg[pos+1:]
    return msg

for i in xrange(bytelength):
